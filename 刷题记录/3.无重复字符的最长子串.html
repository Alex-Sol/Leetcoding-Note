3. 无重复字符的最长子串

给定一个字符串，请你找出其中不含有重复字符的 最长子串 的长度。

示例 1:

输入: "abcabcbb"
输出: 3 
解释: 因为无重复字符的最长子串是 "abc"，所以其长度为 3。

示例 2:

输入: "bbbbb"
输出: 1
解释: 因为无重复字符的最长子串是 "b"，所以其长度为 1。

示例 3:

输入: "pwwkew"
输出: 3
解释: 因为无重复字符的最长子串是 "wke"，所以其长度为 3。
     请注意，你的答案必须是 子串 的长度，"pwke" 是一个子序列，不是子串。
  • 1 滑窗
设chars = s[start,end)

start,end初始化为0。

while(end没超过s的末尾)

​    if chars中有与s[end]相同的字符时：

​        start设为重复位置+1

​        length=end-start

​        结束检查

​    end++

​    length++

输出整个过程中最大的length

class Solution {
public:
    int lengthOfLongestSubstring(string s) {
        int start(0),end(0),length(0),result(0);
        while(end<s.size())
        {
            char tmp = s[end];
            //用for循环检查s[start,end)中是否有元素与s[end]相同
            for(int i=start;i<end;i++)
            {
                if(s[i]==tmp)
                {
                    start = i+1;  //若有相同，则更新start，使得s[start,end）中没有与s[end]重复的字符
                    length = end-start;
                    break;
                }
            }
            end++;
            length++;
            result = max(result,length);
        }
        return result;

    }
};
  • 2 hashmap

思路和1一样，只是把判断[start,end）内是否有与是s[end]重复的字符的任务用hashmap实现。

提交后的runtime和memory usage却还没有1的效果好

class Solution {
public:
    int lengthOfLongestSubstring(string s) {
        int start(0),end(0),length(0),result(0);
        unordered_map<char,int> hash;
        while(end<s.size())
        {
            char tmp = s[end];
            if (hash.find(tmp) != hash.end() //即hash中存在key为tmp
            && hash[tmp]>=start  //由于hash没有擦除start之前的元素，故需要判定重复的字符是否在[start,end)内
            )
            {
                start = hash[tmp]+1;
                length = end-start;
            }
            hash[tmp] = end; //不管tmp在hash中存不存在，记录tmp在目前搜索范围内最靠后的位置

            end++;
            length++;
            result = max(result,length);
        }
        return result;

    }
};
  • 3 数组代替hashmap
class Solution
{
public:
    int lengthOfLongestSubstring(string s)
    {
        //s[start,end) 前面包含 后面不包含
        int start(0), end(0), length(0), result(0);
        int sSize = int(s.size());
        vector<int> vec(128, -1);//所有的char只有128个，故把数组设为128大小，索引是int（char),实值是s中对应的最新的索引
        while (end < sSize)
        {
            char tmpChar = s[end];
            //仅当s[start,end) 中存在s[end]时更新start
            if (vec[int(tmpChar)] >= start)
            {
                start = vec[int(tmpChar)] + 1;
                length = end - start;
            }
            vec[int(tmpChar)] = end;

            end++;
            length++;
            result = max(result, length);
        }
        return result;
    }
};
